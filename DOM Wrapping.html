add single child into signle parent
<div class="boxes">
    <div class="box"></div>
</div>
<div class="box"></div>
<div class="box"></div>
const box = document.querySelector('.box');

// नया parent बनाओ
const wrapper = document.createElement('div');
wrapper.className = 'boxes';

// child के पहले add करो
box.before(wrapper);
// child को parent में add करो
wrapper.appendChild(box);
// you may use wrapper.append(box)
Alternative Method (replaceWith)
const box = document.querySelector(".box");

// create new parent
const parent = document.createElement("div");
parent.className = "boxes";

// replace child into parent
box.replaceWith(parent);
// and add that child into replaced parent
parent.appendChild(box);
add every child into single parent
<div class="boxes">
    <div class="box"></div>
    <div class="box"></div>
    <div class="box"></div>
</div>
const boxes = document.querySelectorAll(".box");
const boxe1 = document.querySelector(".box");

// नया parent बनाओ
const wrapper = document.createElement("div");
wrapper.className = "boxes";

// add parent before first child
boxe1.before(wrapper);

// add all childs into parent
for (const element of boxes) {
    wrapper.appendChild(element);
}
Alternative Method (replaceWith)
const boxes = document.querySelectorAll(".box");
const box1 = document.querySelector(".box");

add every child into its every parent
<div class="boxes">
    <div class="box"></div>
</div>
<div class="boxes">
    <div class="box"></div>
</div>
<div class="boxes">
    <div class="box"></div>
</div>

const boxes = document.querySelectorAll(".box");

for (const element of boxes) {
    // create new parents
    const parent = document.createElement("div");
    parent.className = "boxes";
    // add parents before each child
    element.before(parent);
    // add all childs into all their own parents
    parent.appendChild(element);
}

Alternative Method (replaceWith)
const boxes = document.querySelectorAll(".box");

for (const element of boxes) {
    // create parents
    const parents = document.createElement("div");
    parents.className = "boxes";

    // replace all parents with all child
    element.replaceWith(parents);
    // add all child into all parent
    parents.appendChild(element);
}

Async function syntax

async function fun_name(params) {...}
return promise

Example

async function hello(){
    console.log('hello');
}
hello();
[[Prototype]] : Promise
[[PromiseState]] : "fulfilled"

Await use

function wetherData() {
    return new Promise(function(resolve, reject){
        console.log('getting wether data...')
        setTimeout(function(){
            console.log('got wether data ✅')
            resolve("done ✅");
        },2000);
    })
}
async function getWetherData() {
    await wetherData();
}
getWetherData();
/*
getting wether data...
got wether data ✅
*/

function wetherData(data){
    return new Promise(function(resolve, reject){
        console.log("loading wether data...");
        setTimeout(function(){
            console.log('got wether data ✅');
            resolve('done');
        },2000)
    })
}
async function getWetherData(){
    await wetherData(1);
    await wetherData(2);
    await wetherData(3);
}
getWetherData();
/*
loading wether data...
got wether data ✅
loading wether data...
got wether data ✅
loading wether data...
got wether data ✅
*/

IIFE - Immediately Invoked Function Expression

Syntax

// way 1 normal function
(function(){...})();
// way 2 arrow function
(() => {...})();
// way 3 async function
(async function () {...}) ();

Example

(async function(){
    await wetherData(1);
    await wetherData(2);
    await wetherData(3);
})();

print resolve or reject 

function wetherData(data){
    return new Promise(function(resolve, reject){
        console.log('getting wether data...');
        setTimeout(function(){
            resolve("got wether data ✅");
        },2000)
    })
}
async function getWetherData(){
    let res1 = await wetherData();
    console.log(res1);
    let res2 = await wetherData();
    console.log(res2);
    let res3 = await wetherData();
    console.log(res3);
}
getWetherData();

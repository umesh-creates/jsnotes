🔥 Promises & Promise chain

new Promise ✅
new promise ❌
pending - resolve - reject
             |
         fulfilled
resolve ❌ + reject ❌ = pending
resolve ✅ + reject ❌ = resolve
resolve ❌ + reject ✅ = reject

Syntax

level 0
let promise = new Promise(function(resolve, reject) {
    // some work
});

--> pending

let promise = new Promise((resolve, reject) => {
    console.log("i'm promise"); // i'm promise
})
console.log(promise);
/*
[[Prototype]] : Promise
[[PromiseState]] : "pending"
[[PromiseResult]] : undefined
*/

--> Resolve

let promise = new Promise((resolve,  reject)=> {
    console.log("i'm promise");
    resolve('succeed');
})
console.log(promise);
/*
[[Prototype]] : Promise
[[PromiseState]] : "fulfilled"
[[PromiseResult]] : succeed
*/

--> Reject

let promise = new Promise((resolve, reject) => {
    console.log("i'm promise");
    reject("Error hai");
})

/*

[[PromiseResult]] : "Error hai"

.then() = resolve

without function
let promise = new Promise(function(resolve, reject) {
    resolve("data mil gaya");
});

promise.then(function(result) {
    console.log(result); // data mil gaya
});
✅(A)(with function) Function Promise RETURN करता है
function getData() {
    return new Promise(function(resolve, reject) {
        resolve("data mil gaya");
    });
}

getData().then(function(result) {
    console.log(result); // data mil gaya
});
❌(B) Function Promise RETURN नहीं करता

    new Promise(function(resolve, reject) {
        resolve("data mil gaya");
    });
}

getData().then(function(result) {
    console.log(result);
});

Alternative within Function
function getData() {
    new Promise(function(resolve, reject) {
        resolve("data mil gaya");
    })
    .then(function(res) {
        console.log(res); // data mil gaya
    });
}

.catch() = reject

without function
let promise = new Promise(function(resolve, reject) {
    reject("error aa gaya");
});

promise.catch(function(error) {
    console.log(error); // error aa gaya
});
✅ (A) Return वाला (with function)
function getError() {
    return new Promise(function(resolve, reject) {
        reject("error aa gaya");
    });

    console.log(err); // error aa gaya
});

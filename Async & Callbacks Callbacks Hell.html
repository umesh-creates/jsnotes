🔥 callbacks

Direct calls vs Callbacks

🔹 Step 1: Simple case (दोनों काम कर रहे हैं)

✅ Direct Call

function process() {
    console.log("Start");
    sayHello();
}

function sayHello() {
    console.log("Hello");
}

process();

✅ Callback

function process(fn) {
    console.log("Start");
    fn();
}

function sayHello() {
    console.log("Hello");
}

process(sayHello);
passing arrow function
function sayhi(fn){
    console.log('hi');
    fn();
}

sayhi(()=> {
    console.log('hello')
});

with Arrow Function

const fun = (getData, nextData) => {
    console.log("loading...")
    setTimeout(() => {
        console.log(getData);
        if (nextData){
            nextData();
        }
    }, 2000);
}
fun(1, () => {
    fun(2);
})

This is Callbacks Hell

function getData(dataId, getNextData){
    console.log('loading...')
    setTimeout(() => {
        console.log('data : ',dataId);
        if (getNextData){
            getNextData();
        }
    }, 2000);
}

getData(1, () => {
    getData(2, ()=> {
        getData(3, ()=>{
            getData(4);

    })
